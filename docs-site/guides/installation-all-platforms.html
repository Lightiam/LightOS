Method 1: Homebrew (Recommended)

1 Install Homebrew (if not already installed) 
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
2 Add LightOS tap 
brew tap lightiam/lightos
3 Install LightOS 
# Standard installation
brew install lightos

# With CUDA support (for external NVIDIA eGPU)
brew install lightos --with-cuda
4 Verify installation 
lightos --version
python3 -c "from lightos_accelerated import LightDevice; print('‚úÖ LightOS ready!')"
‚úÖ Installation complete! LightOS is now ready to use.

Method 2: pip (Python Package)

1 Ensure Python 3.8+ is installed 
python3 --version
# Should show Python 3.8 or later

# If not installed:
brew install python@3.11
2 Install LightOS via pip 
pip3 install lightos-accelerated
3 Install optional components 
# For gRPC server
pip3 install lightos-accelerated[server]

# For Jupyter notebooks
pip3 install lightos-accelerated[jupyter]

# Install everything
pip3 install lightos-accelerated[all]

Method 3: PKG Installer

1 Download PKG installer

Visit GitHub Releases and download LightOS-Installer-v1.0.0.pkg

2 Run the installer
  • Double-click the downloaded PKG file
  • If macOS warns "unidentified developer", go to System Preferences ‚Üí Security & Privacy ‚Üí "Open Anyway"
  • Follow the installation wizard
  • Enter administrator password when prompted
3 Verify installation 
lightos --version

Method 4: Build from Source

1 Install dependencies 
brew install cmake ninja python@3.11 libomp
2 Clone repository 
git clone https://github.com/lightiam/lightos.git
cd lightos
3 Build C++ core 
cd inference-subsystem/core
cmake -B build -G Ninja \
  -DCMAKE_BUILD_TYPE=Release \
  -DCMAKE_OSX_ARCHITECTURES="x86_64;arm64"
cmake --build build --parallel
sudo cmake --install build
4 Install Python bindings 
cd ../python-bindings
pip3 install -e .

Method 1: APT (Ubuntu/Debian)

1 Add LightOS repository 
# Add GPG key
curl -fsSL https://packages.lightos.ai/gpg | sudo apt-key add -

# Add repository
echo "deb https://packages.lightos.ai/apt stable main" | sudo tee /etc/apt/sources.list.d/lightos.list

# Update package list
sudo apt update
2 Install LightOS 
# Core package
sudo apt install lightos

# With NVIDIA GPU support
sudo apt install lightos-cuda

# With AMD GPU support
sudo apt install lightos-rocm

# Development headers
sudo apt install lightos-dev

# Documentation
sudo apt install lightos-doc
3 Start systemd service 
# Start LightOS server
sudo systemctl start lightos

# Enable auto-start on boot
sudo systemctl enable lightos

# Check status
sudo systemctl status lightos
4 Verify installation 
lightos --version
python3 -c "from lightos_accelerated import LightDevice; print('‚úÖ Ready!')"

Method 2: pip (Universal)

1 Install Python and dependencies 
# Ubuntu/Debian
sudo apt install python3 python3-pip python3-dev build-essential

# Fedora/RHEL
sudo dnf install python3 python3-pip python3-devel gcc-c++

# Arch Linux
sudo pacman -S python python-pip base-devel
2 Install LightOS 
pip3 install lightos-accelerated

Method 3: AppImage (Portable)

1 Download AppImage 
wget https://github.com/lightiam/lightos/releases/latest/download/LightOS-x86_64.AppImage
chmod +x LightOS-x86_64.AppImage
2 Run LightOS 
# Run directly (no installation needed)
./LightOS-x86_64.AppImage --version

# Or install system-wide
./LightOS-x86_64.AppImage --install

Method 4: Build from Source

1 Install build dependencies 
# Ubuntu/Debian
sudo apt install build-essential cmake ninja-build python3-dev libgomp1 libopenblas-dev

# Fedora/RHEL
sudo dnf groupinstall "Development Tools"
sudo dnf install cmake ninja-build python3-devel openblas-devel
2 Clone and build 
git clone https://github.com/lightiam/lightos.git
cd lightos/inference-subsystem/core

cmake -B build -G Ninja -DCMAKE_BUILD_TYPE=Release
cmake --build build --parallel
sudo cmake --install build

cd ../python-bindings
pip3 install -e .

GPU Support on Linux

NVIDIA NVIDIA GPU Setup 
# Install NVIDIA drivers
sudo apt install nvidia-driver-535

# Install CUDA Toolkit
wget https://developer.download.nvidia.com/compute/cuda/12.3.0/local_installers/cuda_12.3.0_545.23.06_linux.run
sudo sh cuda_12.3.0_545.23.06_linux.run

# Verify
nvidia-smi
nvcc --version
AMD AMD GPU Setup (ROCm) 
# Add ROCm repository
wget -qO - https://repo.radeon.com/rocm/rocm.gpg.key | sudo apt-key add -
echo "deb [arch=amd64] https://repo.radeon.com/rocm/apt/5.7 ubuntu main" | sudo tee /etc/apt/sources.list.d/rocm.list

# Install ROCm
sudo apt update
sudo apt install rocm-dkms

# Verify
rocm-smi

Docker Installation

Run LightOS in a containerized environment (all platforms)

1 Install Docker 
# Linux (Ubuntu/Debian)
curl -fsSL https://get.docker.com | sh
sudo usermod -aG docker $USER

# macOS: Download Docker Desktop
# https://www.docker.com/products/docker-desktop/

# Windows: Download Docker Desktop
# https://www.docker.com/products/docker-desktop/
2 Pull LightOS image 
docker pull lightos/lightos-inference:latest

# Verify image size (should be <700MB)
docker images lightos/lightos-inference
3 Run container 
# CPU-only
docker run -d -p 50051:50051 --name lightos lightos/lightos-inference:latest

# With NVIDIA GPU
docker run -d -p 50051:50051 --gpus all --name lightos lightos/lightos-inference:latest

# With AMD GPU
docker run -d -p 50051:50051 --device=/dev/kfd --device=/dev/dri --name lightos lightos/lightos-inference:latest
4 Verify container 
# Check status
docker ps

# View logs
docker logs lightos

# Interactive Python session
docker exec -it lightos python3

Docker Compose

Create docker-compose.yml:

version: '3.8'

services:
  lightos:
    image: lightos/lightos-inference:latest
    ports:
      - "50051:50051"
    deploy:
      resources:
        reservations:
          devices:
            - driver: nvidia
              count: 1
              capabilities: [gpu]
    environment:
      - LIGHTOS_PORT=50051
      - LIGHTOS_DEVICES=auto
    restart: unless-stopped

Run with:

docker-compose up -d

Cloud Deployment

Deploy LightOS on AWS, Azure, or Google Cloud Platform

AWS EC2

1 Launch EC2 instance 
aws ec2 run-instances \
  --image-id ami-lightos-gpu \
  --instance-type p3.2xlarge \
  --key-name your-key \
  --security-group-ids sg-xxxxxxxx
2 SSH and install 
ssh -i your-key.pem ubuntu@ec2-instance

# Install via pip
pip3 install lightos-accelerated

# Or use Docker
docker pull lightos/lightos-inference:latest

Azure VM

1 Create VM 
az vm create \
  --resource-group lightos-rg \
  --name lightos-vm \
  --image UbuntuLTS \
  --size Standard_NC6 \
  --admin-username azureuser

Google Cloud Platform

1 Create instance with GPU 
gcloud compute instances create lightos-vm \
  --zone=us-central1-a \
  --machine-type=n1-standard-4 \
  --accelerator=type=nvidia-tesla-v100,count=1 \
  --image-family=ubuntu-2204-lts \
  --image-project=ubuntu-os-cloud
